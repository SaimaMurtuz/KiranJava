class incrementDecrement{
public static void main(String args[]){
int i =90;
i++; //91

int j =i+9;// 100
syso(i++); // 92
--j; // 99
syso(j*i);// 99*92=9108
i++; // 93
--i; // 92
syso(--i); //91
++i; // 92
syso(j+3); // 99+3=102

j--; // 98
++j; // 99
syso(j+7); // 99+7 =106
j--; // 98
i++; // 93
--i; // 92

syso(i+j); //92+98=190
syso(i); // 92
syso(j); // 98
}

}
